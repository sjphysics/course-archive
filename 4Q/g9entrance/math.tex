					\question Which one of the following statements is not true? \\
					\begin{oneparchoices}
						\choice The square of a real number is always positive 
						\choice The cube of a real number can be negative \\
						\choice The cube root of a number can be negative 
						\choice The square of a real number can be negative 
					\end{oneparchoices}
					\question Consider an integer $n$, such that $-3<n<7$. What is the probability that when $n$ is selected at random, n is 0? \\
					\begin{oneparchoices}
						\choice $\dfrac{1}{9}$
						\choice $\dfrac{1}{8}$
						\choice $\dfrac{1}{10}$
						\choice $\dfrac{1}{7}$
					\end{oneparchoices} 
					\question A right angled triangle also happens to be isosceles, what are the measures of its interior angle? \\
					\begin{oneparchoices}
						\choice $90^0, 30^0, 60^0$
						\choice $90^0, 45^0, 45^0$
						\choice $60^0, 60^0, 60^0$
						\choice $90^0, 10^0, 80^0$
					\end{oneparchoices}
					\question A quadrilateral is inscribed in a circle. If one of its angles measures $100^0$, what is the value of the angle opposite to it? \\
					\begin{oneparchoices}
						\choice $70^0$
						\choice $80^0$
						\choice $90^0$
						\choice $100^0$
					\end{oneparchoices}
					\question What is the value of $\sqrt{\sqrt{\sqrt{\sqrt{\sqrt{1024}}}}}$?\\
					\begin{oneparchoices}
						\choice 2
						\choice $\sqrt{2}$
						\choice 2$\sqrt{2}$
						\choice $\sqrt{32}$
					\end{oneparchoices}
					\question A 12m high pole casts a shadow of 5m. What is the distance from the top of the pole to the tip of the shadow? \\
					\begin{oneparchoices}
						\choice 12 m
						\choice 13 m
						\choice 5 m
						\choice 17 m
					\end{oneparchoices}	
					\question Which one of the following set of numbers can't be assigned to the lengths of the sides of a right angled triangle? \\
					\begin{oneparchoices}
						\choice 4,3,5
						\choice 6,8,10
						\choice 12,3,15
						\choice 2,2,4
					\end{oneparchoices}
					\question Which of the following is true? \\
					\begin{oneparchoices}
						\choice $\sqrt{a}=a^{2}$
						\choice $\sqrt{a}=a^{\frac{1}{3}}$
						\choice ${a^{x}}^y=a^{xy}$
						\choice $a^xa^y=a^{\frac{y}{x}}$
					\end{oneparchoices}
					\question For the expression $\dfrac{2y}{x}$, what are the limitations we have? \\
					\begin{oneparchoices}
						\choice $x\not= y$
						\choice $y\not= 0$
						\choice $x\not= 0$
						\choice $x\not= 1$
					\end{oneparchoices}
					\question If $a^{\frac{-1}{2}}=x$ , where $a$ > 0 and $x$ > 0, which of the following equations gives $a$
					in terms of $x$? \\
					\begin{oneparchoices}
						\choice $a=\dfrac{1}{\sqrt{x}}$
						\choice $a=\dfrac{1}{x^2}$
						\choice $a=\sqrt{x}$
						\choice $a=-{x}^2$
					\end{oneparchoices}
					\question A farmer can pick 12 cabbages in 1 hour. Working at the
					same rate, how long in hours would it take two farmers to pick 48 cabbages?\\
					\begin{oneparchoices}
						\choice 1
						\choice 2
						\choice 6
						\choice 4
						\choice 8
					\end{oneparchoices}
					\question Which of the following is NOT a positive multiple of 9 + 3?\\
					\begin{oneparchoices}
						\choice 3
						\choice 12
						\choice 24
						\choice 48
						\choice 60
					\end{oneparchoices}
					\question Lydia can run at a pace of 6 miles per hour. Running at the same rate, how many miles can she run in 90 minutes?\\
					\begin{oneparchoices}
						\choice 4
						\choice 6
						\choice 8
						\choice 9
						\choice 12
					\end{oneparchoices}
					\question Two consecutive integers $m$ and $n$ are prime numbers. Which of the following is equal to $mn$? \\
					\begin{oneparchoices}
						\choice 1
						\choice 2
						\choice 6
						\choice 9
						\choice 15
					\end{oneparchoices}
					\question If $x + 1 = 23$, what is the value of $3x + 3$? \\
					\begin{oneparchoices}
						\choice 22
						\choice 46
						\choice 66
						\choice 69
						\choice 72
					\end{oneparchoices}
				\end{questions}
